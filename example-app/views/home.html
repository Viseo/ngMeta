Define meta tags in your app config for the routes/states of your Angular single-page application.

Supports ui-router and angular-route (ngRoute)

Getting Started

  1. Download ngMeta 

            bower install --save ngMeta
    Check out the download section for other download options (NPM, JS file)

  2. Add 'ngMeta' as a project dependency 

            angular.module('yourApp', [....,'ngMeta']);

  3. Add meta tags to your routes/states 

            angular.module('yourApp', [....,'ngRoute','ngMeta'])
        .config(function($routeProvider) {
          $routeProvider
          .when('/', {
            templateUrl: 'views/home.html',
            controller: 'HomeCtrl',
            data: {
              meta: {
                'title': 'Home page',
                'description': 'Home page description'
              }
            }
          })
          .when('/about', {
            templateUrl: 'views/about.html',
            controller: 'AboutCtrl',
            data: {
              meta: {
                'title': 'About Us',
                'description': 'About us description'
              }
            }
          })
          ...
        });

  4. Initialize ngMeta using the run block 

            angular.module('yourApp', [....,'ngMeta'])
        .config(function(...) {
          ...
        })
        .run(function(ngMeta) {
          ngMeta.init();
        });

  5. Use meta tags in your HTML file 

            
        <head>
        <title>{{ngMeta.title}}</title>
        <meta property="og:title" content="{{ngMeta.title}}" />
        <meta property="og:description" content="{{ngMeta.description}}" />
        </head>
